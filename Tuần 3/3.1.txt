1) Viết chương trình tính UCLN, BCNN 2 số nguyên a và b
#include<iostream>

using namespace std;

int GCD(int a, int b) {
	int min_val = a < b ? a : b;
	for (int i = min_val; i >= 1; i--) {
		if (a % i == 0 && b % i == 0) return i;
	}
}

int LCM(int a, int b) {
	return a * b / GCD(a, b);
}
int main() {
	int a, b;
	cout << "Enter a: ";
	cin >> a;
	cout << "Enter b: ";
	cin >> b;
	cout << "The greatest common divisor: " << GCD(a, b) << endl;
	cout << "The least common multiple: " << LCM(a, b);
	return 0;
}

#include<iostream>

using namespace std;

int MAX(int n) {
	int a;
	int max = INT_MIN;
	for (int i = 1; i <= n; i++) {
		cout << "Number " << i << ": ";
		cin >> a;
		if (a > max) max = a;
	}
	return max;
}
int main() {
	int n = 0;
	do {
		system("cls");
		cout << "Enter n: ";
		cin >> n;
		if (n < 0) {
			cout << "Invalid number";
			system("pause");
		}
		else {
			cout << "The greatest number is: " << MAX(n);
		}
	} while (n < 0);
	return 0;
}

#include<iostream>

using namespace std;

void hoanvi(int& a, int& b) {
	int temp = a;
	a = b;
	b = temp;
}
int main() {
	int a, b;
	cout << "Enter a: ";
	cin >> a;
	cout << "Enter b: ";
	cin >> b;
	hoanvi(a, b);
	cout << "AFTER SWAPPING" << endl;

}
